Linked list - Delete second last

Write a Program to delete the second last element from the list. If deletion of second last element is not possible, print "Deletion of second last element is not possible", Otherwise print the list after deleting the second last element.

Input Format

Read elements upto -1

Constraints

1<=n<=10^7

Output Format

print output

Sample Input 0

1
2
3
4
-1
Sample Output 0

Before deleting:
1 2 3 4
After deleting:
1 2 4
Sample Input 1

23
34
55
12
66
45
33
-1
Sample Output 1

Before deleting:
23 34 55 12 66 45 33
After deleting:
23 34 55 12 66 33


class Node:
    def __init__(self,x):
        self.data=x
        self.next=None
class Deletion:
    def __init__(self):
        self.head=self.tail=None
    def create(self,x):
        n=Node(x)
        if self.head==None:
            self.head=self.tail=n
        else:
            self.tail.next=n
            self.tail=n
    def show(self):
        temp=self.head
        c=0
        a=[]
        while(temp!=None):
            a.append(temp.data)
            temp=temp.next
            c=c+1
        print("Before deleting:")
        print(*a)
        print("After deleting:")
        a.remove(a[c-2])
        print(*a)
            
            
d1=Deletion()
while True:
    x=int(input())
    if x==-1:
        break
    else:
        d1.create(x) 
d1.show()
